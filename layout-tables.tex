%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%  Creating tables

% arg1 = position
% arg2 = column definition string (eg |cc|r|lp{2cm}||)
% arg3 = column headings
% arg4 = body of table
% arg5 = caption
% arg6 = label for the figure
%
\newcommand{\tab}[6]{%
  \begin{table}[#1]
    \centering
      \caption{\label{#6}#5}%
      \begin{tabular}{#2}%
        \hline #3 \hline%
        #4%
        \hline%
      \end{tabular}%
  \end{table}
}
